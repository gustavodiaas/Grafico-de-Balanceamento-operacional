    <script>
        // Inicializa ícones
        lucide.createIcons();

        // Estado da Aplicação
        let operations = [];
        let taktTime = 0;

        // Função de Tema (Dark/Light)
        function toggleTheme() {
            document.documentElement.classList.toggle('dark');
        }

        // Cálculo do Takt Time
        function calculate() {
            const shift = parseFloat(document.getElementById('shiftTime').value) || 0;
            const demand = parseFloat(document.getElementById('demand').value) || 0;

            if (shift > 0 && demand > 0) {
                // Converte horas para minutos e divide
                taktTime = (shift * 60) / demand;
                
                document.getElementById('taktResult').classList.remove('hidden');
                document.getElementById('taktValue').innerText = taktTime.toFixed(2);
                renderChart(); // Atualiza gráfico se mudar o Takt
            } else {
                taktTime = 0;
                document.getElementById('taktResult').classList.add('hidden');
            }
        }

        // Adicionar Operação
        function addOperation() {
            const nameInput = document.getElementById('opName');
            const timeInput = document.getElementById('opTime');
            
            const name = nameInput.value.trim();
            const time = parseFloat(timeInput.value);

            if (!name || isNaN(time) || time <= 0) {
                alert("Por favor, preencha o nome e um tempo válido.");
                return;
            }

            operations.push({ id: Date.now(), name, time });
            
            // Limpa campos
            nameInput.value = '';
            timeInput.value = '';
            nameInput.focus();

            renderAll();
        }

        // Remover Operação
        function removeOperation(id) {
            operations = operations.filter(op => op.id !== id);
            renderAll();
        }

        // Renderiza tudo (Tabela e Gráfico)
        function renderAll() {
            renderTable();
            renderChart();
        }

        // Renderiza Tabela
        function renderTable() {
            const tbody = document.getElementById('opList');
            tbody.innerHTML = '';

            operations.forEach(op => {
                const isGargalo = taktTime > 0 && op.time > taktTime;
                const statusHtml = isGargalo 
                    ? `<span class="px-2 py-1 bg-red-100 text-red-700 rounded text-xs font-bold">Gargalo</span>`
                    : `<span class="px-2 py-1 bg-green-100 text-green-700 rounded text-xs">OK</span>`;

                const tr = document.createElement('tr');
                tr.className = "border-b dark:border-gray-700 last:border-0";
                tr.innerHTML = `
                    <td class="py-3 font-medium">${op.name}</td>
                    <td class="py-3">${op.time.toFixed(2)} min</td>
                    <td class="py-3">${statusHtml}</td>
                    <td class="py-3 text-right">
                        <button onclick="removeOperation(${op.id})" class="text-red-500 hover:text-red-700">
                            <i data-lucide="trash-2" class="w-4 h-4"></i>
                        </button>
                    </td>
                `;
                tbody.appendChild(tr);
            });
            lucide.createIcons(); // Recarrega ícones novos
        }

        // Renderiza Gráfico (Manual com CSS)
        function renderChart() {
            const container = document.getElementById('chartContainer');
            const emptyState = document.getElementById('emptyState');
            const taktLine = document.getElementById('taktLine');

            // Remove barras antigas (mantém emptyState e taktLine)
            Array.from(container.children).forEach(child => {
                if (child.id !== 'emptyState' && child.id !== 'taktLine') {
                    container.removeChild(child);
                }
            });

            if (operations.length === 0) {
                emptyState.style.display = 'flex';
                taktLine.classList.add('hidden');
                return;
            }

            emptyState.style.display = 'none';

            // Define altura máxima (Maior tempo ou Takt time + margem)
            const maxTime = Math.max(...operations.map(o => o.time), taktTime * 1.2 || 0);

            // Desenha linha do Takt
            if (taktTime > 0) {
                taktLine.classList.remove('hidden');
                const taktHeightPercent = (taktTime / maxTime) * 100;
                taktLine.style.bottom = `${taktHeightPercent}%`;
            } else {
                taktLine.classList.add('hidden');
            }

            // Desenha Barras
            operations.forEach(op => {
                const heightPercent = (op.time / maxTime) * 100;
                const isGargalo = taktTime > 0 && op.time > taktTime;
                const colorClass = isGargalo ? 'bg-red-500' : 'bg-blue-400';

                const bar = document.createElement('div');
                bar.className = "flex flex-col items-center justify-end h-full flex-1 group relative";
                bar.innerHTML = `
                    <div class="w-full max-w-[50px] ${colorClass} rounded-t transition-all duration-500 hover:opacity-80 relative" style="height: ${heightPercent}%">
                        <span class="absolute -top-6 left-1/2 -translate-x-1/2 text-xs font-bold opacity-0 group-hover:opacity-100 transition-opacity">
                            ${op.time}
                        </span>
                    </div>
                    <span class="text-[10px] text-gray-500 mt-1 truncate w-full text-center">${op.name}</span>
                `;
                container.appendChild(bar);
            });
        }

        // Exportar PDF
        window.jsPDF = window.jspdf.jsPDF;

        async function exportPDF() {
            const element = document.getElementById('reportArea');
            const btn = document.querySelector('button[onclick="exportPDF()"]');
            
            // Feedback visual
            const originalText = btn.innerHTML;
            btn.innerHTML = "Gerando PDF...";
            btn.disabled = true;

            try {
                // Força tema claro para o PDF ficar bonito
                const wasDark = document.documentElement.classList.contains('dark');
                if (wasDark) document.documentElement.classList.remove('dark');

                const canvas = await html2canvas(element, { scale: 2 });
                const imgData = canvas.toDataURL('image/png');
                
                const pdf = new jsPDF('p', 'mm', 'a4');
                const pdfWidth = pdf.internal.pageSize.getWidth();
                const pdfHeight = (canvas.height * pdfWidth) / canvas.width;

                pdf.setFontSize(18);
                pdf.text("Relatório GBO", 10, 15);
                pdf.setFontSize(10);
                pdf.text(`Gerado em: ${new Date().toLocaleDateString()}`, 10, 20);

                pdf.addImage(imgData, 'PNG', 0, 30, pdfWidth, pdfHeight);
                pdf.save(`gbo-relatorio-${Date.now()}.pdf`);

                // Restaura tema
                if (wasDark) document.documentElement.classList.add('dark');

            } catch (err) {
                console.error(err);
                alert("Erro ao gerar PDF");
            } finally {
                btn.innerHTML = originalText;
                btn.disabled = false;
            }
        }

        // Roda cálculo inicial
        calculate();
    </script>
